// Dériver un formulaire depuis un modèle
#[form(schema = eihwaz_users_schema,
       fields = [username, email, password])]
pub struct RegisterForm;

// clean() = override officiel du trait RuniqueForm
#[async_trait]
impl RuniqueForm for RegisterForm {
    impl_form_access!(model);

    async fn clean(&mut self) -> Result<(), StrMap> {
        let mut errors = StrMap::new();
        if self.get_string("username").len() < 3 {
            errors.insert("username", "3 caractères minimum");
        }
        if !self.get_string("email").contains('@') {
            errors.insert("email", "adresse email invalide");
        }
        if self.get_string("password").len() < 10 {
            errors.insert("password", "10 caractères minimum");
        }
        if errors.is_empty() { Ok(()) } else { Err(errors) }
    }
}

views.rs
pub async fn soumission_inscription(
    mut request: Request,
    Prisme(mut form): Prisme<RegisterForm>,
) -> AppResult<Response> {
    if request.is_post() && form.is_valid().await {
        match form.save(&db).await {
            Ok(user) => {
                auth_login(&session, user.id, &user.username).await;
                return Ok(Redirect::to("/profil").into_response());
            }
            Err(err) => form.get_form_mut().database_error(&err),
        }
    }
    request.render("inscription_form.html")
}
