Déclarer un modèle avec la macro model!, générer et appliquer les migrations SeaORM.

Déclarer un modèle — macro model!
// src/models/article.rs
#[model(
    fields = [title, content, published, author_id],
    belongs_to = [User [cascade = false]]
)]
pub struct Article;

// Champs avec contraintes
#[model(fields = [
    username: { unique = true, max_length = 150 },
    email:    { unique = true },
    bio:      { null = true, blank = true },
])]
pub struct Profil;
Générer les migrations
# Génère les fichiers de migration depuis les entités
$ runique makemigrations \
    --entities src/entities \
    --migrations migration/src

# Fichier généré : migration/src/m20240101_000001_create_article.rs
Appliquer / Rollback
# Appliquer toutes les migrations en attente
$ runique migration up

# Rollback de la dernière migration
$ runique migration down

# Voir l'état des migrations
$ runique migration status
