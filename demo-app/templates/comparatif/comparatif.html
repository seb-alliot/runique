class ArticleForm(ModelForm):
    class Meta:
        model = Article
        fields = ["title", "content"]

    def clean(self):
        if len(self.cleaned_data["title"]) < 5:
            raise ValidationError("5 car. min.")

// Runique
#[form(schema = article, fields = [title, content])]
pub struct ArticleForm;

async fn clean(&mut self) -> Result<(), StrMap> {
    if self.get_string("title").len() < 5 {
        errors.insert("title", "5 car. min.");
    }
}
