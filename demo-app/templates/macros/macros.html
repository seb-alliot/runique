Macros utilitaires fournies par Runique.

context_update! — injecter des données dans le template
pub async fn ma_vue(mut request: Request) -> AppResult<Response> {
    context_update!(request => {
        "title"       => "Ma page",
        "description" => "Un framework web en Rust",
    });
    request.render("ma_page.html")
}
context! — créer un contexte indépendant
use runique::prelude::*;

// Utile pour construire un contexte sans Request
let ctx = context! {
    "title" => "Ma page",
    "user"  => &username,
};
impl_from_error! — convertir une erreur en AppError
use runique::prelude::*;

// Génère impl From<DbErr> for AppError
// et impl From<anyhow::Error> for AppError
impl_from_error!(
    anyhow::Error => from_anyhow,
    DbErr         => database
);

// Permet d'utiliser ? directement dans les handlers
pub async fn ma_vue(request: Request) -> AppResult<Response> {
    let items = MonModel::objects()
        .all(&db).await?; // DbErr converti automatiquement
    request.render("liste.html")
}
impl_objects! — syntaxe Django-like pour SeaORM
// Généré automatiquement par `runique start` sur chaque entité.
// Ajoute un champ `objects` de style Django — ne change pas
// la quantité de code, juste la syntaxe.

impl_objects!(Entity);

// Équivalences :
Entity::objects.all(&db).await?
// == Entity::find().all(&db).await?

Entity::objects.filter(Column::Title.eq("Rust")).all(&db).await?
// == Entity::find().filter(...).all(&db).await?

Entity::objects.get(1, &db).await?
// == Entity::find_by_id(1).one(&db).await?
