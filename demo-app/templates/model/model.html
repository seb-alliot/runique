    table: "blog",

        title:   String [required],
        email:   String [required],

        summary: String [required],

    }

entities/contribution.rs — relation + contraintes
model! {
    Contributions,
    table: "contributions",
    pk: id => i32,
    fields: {
        user_id:    i32      [required],
        title:      String   [required, max_len(200)],
        content:    String   [required],
        created_at: datetime [auto_now],
    },
    relations: {
        belongs_to: EihwazUsers via user_id [cascade],
    }
}

migration — générer et appliquer
# détecter les changements et générer
runique makemigrations

# appliquer
runique migration up

# lister les rollbacks disponibles
runique migration status

# annuler — par fichier ou par batch
runique migration down --files users/20240101_120000
runique migration down --batch 20240101_120000

# ⚠ rollback partiel : risque de désynchronisation
#   entre les fichiers applied/ et l'état réel de la DB.
#   Toujours rollback dans l'ordre inverse (plus récent d'abord).
