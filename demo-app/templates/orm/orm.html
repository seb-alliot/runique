Requêtes SeaORM via la syntaxe Django-like de Runique — objects, filtres, relations, pagination.

Requêtes de base
use runique::prelude::*;

// Tous les enregistrements
Article::objects.all(&db).await?;

// Par ID
Article::objects.get(1, &db).await?;

// Filtrer
Article::objects
    .filter(Column::Published.eq(true))
    .all(&db).await?;

// Trier
Article::objects
    .order_by(Column::CreatedAt, Order::Desc)
    .all(&db).await?;
Pagination
let page = 1_u64;
let per_page = 10_u64;

let paginator = Article::objects
    .filter(Column::Published.eq(true))
    .paginate(&db, per_page);

let articles = paginator.fetch_page(page - 1).await?;
let total_pages = paginator.num_pages().await?;
Créer / Modifier / Supprimer
// Créer
let article = article::ActiveModel {
    title: Set("Mon article".to_string()),
    published: Set(false),
    ..Default::default()
};
article.insert(&db).await?;

// Modifier
let mut article: article::ActiveModel = found.into();
article.published = Set(true);
article.update(&db).await?;

// Supprimer
found.delete(&db).await?;
Relations (belongs_to)
// Charger les commentaires d'un article
let commentaires = article
    .find_related(commentaire::Entity)
    .all(&db).await?;

// Via ModelForm — cascade déclaré dans model!
#[model(
    fields = [title, content, author_id],
    belongs_to = [
        User [cascade = false]
    ]
)]
pub struct Article;
