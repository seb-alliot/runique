formulaire/image.rs
pub struct ImageForm {
    pub form: Forms,
}

impl RuniqueForm for ImageForm {
    fn register_fields(form: &mut Forms) {
        // Le chemin d'upload vient de .env via StaticConfig
        let config = StaticConfig::from_env();
        form.field(
            &FileField::image("image")
                .label("Choisissez une image")
                .upload_to(&config)
                .max_size_mb(5)
                .max_files(3)
                .max_dimensions(1920, 1080)
                .allowed_extensions(
                    vec!["png", "jpg", "jpeg", "gif"]
                ),
        );
    }
    impl_form_access!();
}
3 façons de configurer le chemin
// 1 — chemin direct (le plus simple)
FileField::image("avatar")
    .upload_to("media/avatars")

// 2 — lit MEDIA_ROOT depuis .env
FileField::image("img")
    .upload_to_env()

// 3 — depuis une StaticConfig existante
let config = StaticConfig::from_env();
FileField::image("img")
    .upload_to(&config)
extensions disponibles
FileField::image("img")    // jpg jpeg png gif webp avif
FileField::document("doc") // pdf doc docx odt
FileField::any("f")       // pas de filtre

FileField::any("data")
    .allowed_extensions(vec!["csv", "json"])
