Déclarer un modèle dans l'admin via la macro admin!, puis générer le panneau avec runique start.

src/admin.rs — déclaration
use crate::entities::{blog, users};
use crate::formulaire::{BlogForm, RegisterForm};

admin! {
    users: users::Model => RegisterForm {
        title: "Utilisateurs",
        permissions: ["admin"]
    }
    blog: blog::Model => BlogForm {
        title: "Articles",
        permissions: ["admin"]
    }
}
Structure du fichier
// nom    : identifiant de la ressource (URL admin)
// Model  : entité SeaORM
// Form   : formulaire utilisé pour create/update
// title  : libellé affiché dans l'interface
// permissions : rôles autorisés à accéder

admin! {
    nom: entities::Model => MonForm {
        title: "Mon modèle",
        permissions: ["admin"]
    }
}
