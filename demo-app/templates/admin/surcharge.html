Remplacer les templates par défaut de l'admin par les vôtres. (fonctionnalité beta)

Surcharge dans main.rs
builder::new(config)
    .with_admin(|a| {
        a.site_title("Administration")
         .auth(RuniqueAdminAuth::new())
         .routes(admins::routes("/admin"))
         .templates(|t| {
             t.with_dashboard("admin/mon_dashboard.html")
         })
         .with_state(admins::admin_state())
    })

Clés injectées par Runique (obligatoires)
// Votre template doit étendre le template admin de base.
// Ces clés sont injectées automatiquement par Runique.
{% raw %}{% extends "admin/admin_template" %}{% endraw %}

{% raw %}{{ site_title }}{% endraw %}       // titre configuré dans .site_title()
{% raw %}{{ site_url }}{% endraw %}         // URL de base
{% raw %}{{ resources }}{% endraw %}        // liste des ressources (key, title, permissions)
{% raw %}{{ resource_counts }}{% endraw %}   // nb d'entrées par ressource
{% raw %}{{ lang }}{% endraw %}             // code langue actif (ex: "fr")

{% raw %}{% block title %}{% endblock %}{% endraw %}       // titre de l'onglet
{% raw %}{% block breadcrumb %}{% endblock %}{% endraw %}  // fil d'ariane
{% raw %}{% block content %}{% endblock %}{% endraw %}     // contenu principal
{% raw %}{% block sidebar %}{% endblock %}{% endraw %}     // barre latérale
