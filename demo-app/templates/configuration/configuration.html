Configuration complète d'une application Runique via le builder — tout est déclaratif, typé, et validé au démarrage.

main.rs — configuration minimale
use runique::app::builder::RuniqueAppBuilder as builder;
use runique::prelude::*;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    init_logging();

    let db_config = DatabaseConfig::from_env()?.build();
    let db = db_config.connect().await?;

    builder::new(RuniqueConfig::from_env())
        .with_database(db)
        .routes(url::routes())
        .statics()
        .build().await?
        .run().await?;
    Ok(())
}
Sécurité — middlewares via builder
use runique::middleware::security::csp::SecurityPolicy;

// .env — redirection HTTPS + HSTS
ENFORCE_HTTPS=true

// main.rs — hosts + CSP + RateLimiter
builder::new(RuniqueConfig::from_env())
    .middleware(|m| m
        // Hôtes autorisés
        .with_allowed_hosts(|h| h
            .enabled(true)
            .host("monsite.fr")
            .host("www.monsite.fr"))
        // CSP — profil strict
        .with_csp(|c| c.policy(SecurityPolicy::strict())))

// RateLimiter — utilisé dans le handler, pas le builder
static LIMITER: LazyLock<RateLimiter> = LazyLock::new(||
    RateLimiter::new()
        .max_requests(100)
        .retry_after(60));

LoginGuard — anti brute-force
use runique::middleware::auth::login_guard::LoginGuard;

// Utilisé dans le handler login, après Prisme
static GUARD: LazyLock<LoginGuard> = LazyLock::new(||
    LoginGuard::new()
        .max_attempts(5)
        .lockout_secs(300));

let key = LoginGuard::effective_key(&username, &ip);
if GUARD.is_locked(&key) {
    return (StatusCode::TOO_MANY_REQUESTS, "Compte temporairement bloqué")
        .into_response();
}
GUARD.record_failure(&key);
