Sessions en mémoire avec nettoyage périodique automatique des sessions expirées.

Configuration dans main.rs
use runique::app::builder::RuniqueAppBuilder as builder;
use runique::prelude::RuniqueConfig;

builder::new(RuniqueConfig::from_env())
    .middleware(|m| {
        m
            // Taille max par session / taille totale du store
            .with_session_memory_limit(
                5 * 1024 * 1024,   // 5 Mo par session
                10 * 1024 * 1024,  // 10 Mo total
            )
            // Intervalle de nettoyage des sessions expirées (en minutes)
            .with_session_cleanup_interval(5)
    })
Lecture / écriture en handler
use runique::prelude::*;

pub async fn mon_handler(mut request: Request) -> AppResult<Response> {
    // Vérifier si l'utilisateur est connecté
