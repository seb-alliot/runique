Configuration de la base de données via variables d'environnement ou URL directe.

main.rs — utilisation dans cette démo
let db_config = DatabaseConfig::from_env()?
    .min_connections(2)
    .build();

let db: DatabaseConnection = db_config.connect().await?;
.env — URL directe (prioritaire)
DB_URL=postgres://user:password@localhost:5432/mabase

# SQLite
DB_URL=sqlite://local_base.sqlite
.env — variables composantes
DB_ENGINE=postgres   # postgres | mysql | mariadb | sqlite
DB_USER=monuser
DB_PASSWORD=secret
DB_HOST=localhost
DB_PORT=5432
DB_NAME=mabase
from_url — URL directe dans le code
DatabaseConfig::from_url("postgres://user:pass@localhost/db")?
    .max_connections(50)
    .build();
Pool de connexions
// Variables .env — valeurs par défaut
DB_MAX_CONNECTIONS=100   // taille max du pool
DB_MIN_CONNECTIONS=20    // taille min maintenue

// Ou via builder
.max_connections(100)
.min_connections(2)
Timeouts
DB_CONNECT_TIMEOUT=2     // connexion (secondes)
DB_ACQUIRE_TIMEOUT=500  // attente pool (ms)
DB_IDLE_TIMEOUT=300     // durée de vie idle (s)
DB_MAX_LIFETIME=3600    // durée de vie max (s)

// Logging SQL (Le système va évoluer)
DB_LOGGING=false
