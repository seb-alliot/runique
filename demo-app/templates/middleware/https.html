Redirection automatique HTTP → HTTPS et en-têtes de sécurité HSTS. Fonctionne via le header X-Forwarded-Proto — nécessite un reverse proxy.

main.rs — activation
use runique::middleware::https::enforce_https;

RuniqueApp::new()
    .middleware(enforce_https())
    .urls(routes())
    .run().await

// Redirige vers HTTPS si X-Forwarded-Proto: http
// Ajoute automatiquement le header HSTS.
En-têtes générés
// Redirection HTTP → HTTPS :
HTTP/1.1 301 Moved Permanently
Location: https://monsite.fr/...

// Header HSTS ajouté sur les réponses HTTPS :
Strict-Transport-Security:
  max-age=31536000; includeSubDomains
Prérequis — reverse proxy
// enforce_https lit X-Forwarded-Proto.
// Ce header doit être injecté par le proxy.

# nginx
proxy_set_header X-Forwarded-Proto $scheme;

# Caddy
# injecté automatiquement

// ⚠ Sans proxy de confiance, ce header
//   peut être forgé par un client.
//   Ne jamais l'activer en local pur.
Combinaison recommandée
// En production, combiner :
RuniqueApp::new()
    .middleware(enforce_https())
    .middleware(allowed_hosts_middleware())
    .middleware(SecurityPolicy::strict())
    .urls(routes())
    .run().await
