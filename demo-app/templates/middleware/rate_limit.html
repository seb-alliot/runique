{{ title }}

Limitation de requêtes par IP avec fenêtre glissante. Répond 429 Too Many Requests quand la limite est dépassée.

Utilisation dans un handler
use runique::prelude::*;

static LIMITER: LazyLock<RateLimiter> = LazyLock::new(|| {
    RateLimiter::new()
        .max_requests(100)
        .window_secs(60)
});

pub async fn login(/* ... */) -> impl IntoResponse {
    if !LIMITER.is_allowed(&ip) {
        return StatusCode::TOO_MANY_REQUESTS.into_response();
    }
    // ...
}
Exemples de configuration
// 5 requêtes par minute
RateLimiter::new().max_requests(5).window_secs(60)

// 3 requêtes par 5 minutes
RateLimiter::new().max_requests(3).window_secs(300)

// 100 requêtes par minute (défaut : 60/60)
RateLimiter::new().max_requests(100).window_secs(60)
Fonctionnement
// Fenêtre glissante par adresse IP.
// Compteur réinitialisé après `window_secs` secondes.

// Réponse quand la limite est dépassée :
HTTP/1.1 429 Too Many Requests
Retry-After: 42

// Le header Retry-After indique
// le délai avant la prochaine fenêtre.
