Protection anti brute-force sur les tentatives de connexion. Lockout par username (connecté) ou par IP anonyme (non connecté).

Utilisateur connecté — clé par username
use runique::prelude::*;

static GUARD: LazyLock<LoginGuard> = LazyLock::new(|| {
    LoginGuard::new()
        .max_attempts(5)
        .lockout_secs(300)
});

pub async fn login(/* ... */) -> impl IntoResponse {
    // username soumis → clé par compte ciblé
    let key = LoginGuard::effective_key(&username, &ip);

    if GUARD.is_locked(&key) {
        let remaining = GUARD.remaining_lockout_secs(&key).unwrap_or(0);
        return (StatusCode::TOO_MANY_REQUESTS,
                format!("Réessayez dans {remaining}s")).into_response();
    }
    match authenticate(&username, &password, &db).await {
        Some(user) => { GUARD.record_success(&key); /* ... */ }
        None       => { GUARD.record_failure(&key); /* ... */ }
    }
}
Utilisateur anonyme — clé par IP
// username vide ou absent → "anonym:{ip}"
// Chaque IP a son propre compteur.
// Bloquer une IP n'affecte pas les autres.

let username = form.username();   // ""
let key = LoginGuard::effective_key(&username, &ip);
// key == "anonym:1.2.3.4"

// username connu → key == "alice"
let key = LoginGuard::effective_key("alice", &ip);
// key == "alice"
Combinaison avec RateLimiter
// RateLimiter  — volume par IP, toutes routes
// LoginGuard   — lockout par compte ou IP anonyme

static IP_LIMITER: LazyLock<RateLimiter> = LazyLock::new(||
    RateLimiter::new().max_requests(10).window_secs(60));
static GUARD: LazyLock<LoginGuard> = LazyLock::new(||
    LoginGuard::new().max_attempts(5).lockout_secs(300));

pub async fn login(/* ... */) -> impl IntoResponse {
    if !IP_LIMITER.is_allowed(&ip) { /* 429 */ }
    let key = LoginGuard::effective_key(&username, &ip);
    if GUARD.is_locked(&key) { /* 429 */ }
    // ...
}
