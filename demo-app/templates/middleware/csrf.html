Le middleware CSRF est activé par défaut dans le builder Runique. Dans les formulaires, la protection est entièrement transparente via Prisme.

main.rs — activation
RuniqueApp::new()
    .urls(routes())
    .run().await

// Le middleware CSRF est inclus automatiquement.
// Aucune configuration manuelle requise.
template — tous les champs d'un coup
{# Rendu de l'ensemble du formulaire #}
<form method="post">
    {% form.inscription_form %}
    <button type="submit">S'inscrire</button>
</form>
template — champ par champ
{# Rendu individuel pour contrôler l'ordre ou la mise en page #}
<form method="post">
    {% form.inscription_form.username %}
    {% form.inscription_form.email %}
    {% form.inscription_form.password %}
    <button type="submit">S'inscrire</button>
</form>

views.rs — extraction avec Prisme
pub async fn inscription(
    mut request: Request,
    Prisme(mut form): Prisme<RegisterForm>,
) -> AppResult<Response> {
    if request.is_post() && form.is_valid().await {
        // Prisme valide le token CSRF avant d'extraire le formulaire.
        // En cas d'échec : redirection silencieuse vers la page courante.
        form.save(&request.engine.db).await?;
    }
    request.render("auth/inscription_form.html")
}
