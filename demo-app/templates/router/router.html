url.rs — déclarer les routes
pub fn routes() -> Router {
    urlpatterns! {
        // Route simple
        "/" => view!{ views::index }, name = "index",

        // Paramètre de chemin
        "/blog/{id}" => view!{ views::blog_detail }, name = "blog_detail",

        // Plusieurs routes groupées
        "/login"  => view!{ views::login },  name = "login",
        "/logout" => view!{ views::logout }, name = "logout",
    }
}

pub async fn blog_detail(
    Path(id): Path<i32>,

) -> AppResult<Response> {

    let article = BlogEntity::find_by_id(id)

views.rs — query param
pub async fn search(
    mut request: Request,
) -> AppResult<Response> {
    // récupère ?username=foo dans l'URL
    let username = request
        .query_param::<String>("username")
        .unwrap_or_default();

    request.render("profile/view_user.html")
}

GET vs POST — même route
pub async fn ma_vue(
    mut request: Request,
) -> AppResult<Response> {
    if request.is_get() {
        return request.render("form.html");
    }
    if request.is_post() {
        // traiter la soumission
        return Ok(Redirect::to("/").into_response());
    }
    request.render("form.html")
}
